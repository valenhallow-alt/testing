-- === Main tab with safe Auto Buy Egg toggle (event-driven + conservative fallback) ===
do
    local MainTab = Window:CreateTab("Main", 4483362458)
    local MainSection = MainTab:CreateSection("Main")

    -- Global flag used by the toggle loop
    _G.UnHub_AutoBuyEgg = _G.UnHub_AutoBuyEgg or false

    -- Internal state
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    local Players = game:GetService("Players")
    local LocalPlayer = Players.LocalPlayer
    local LastEgg = nil -- { id=..., rarity=..., variety=... }
    local purchaseRemote = nil
    local Conduit = nil
    local loopRunning = false

    -- Locate the purchase remote at the path you provided
    local function getPurchaseRemote()
        local ok, inst = pcall(function()
            return ReplicatedStorage
                :WaitForChild("Modules", 2)
                :WaitForChild("Internals", 2)
                :WaitForChild("Skeleton", 2)
                :WaitForChild("Conduit", 2)
                :WaitForChild("Instances", 2)
        end)
        if not ok or not inst then
            warn("[UnHub] getPurchaseRemote: couldn't find Conduit Instances path")
            return nil
        end
        local rem = inst:FindFirstChild("_purchaseEgg")
        if rem then
            print("[UnHub] getPurchaseRemote: found _purchaseEgg remote")
        else
            warn("[UnHub] getPurchaseRemote: _purchaseEgg remote not found in Instances")
        end
        return rem
    end

    -- Attempt to require the Skeleton module and attach Conduit listeners (lightweight)
    pcall(function()
        local ok, skeleton = pcall(function()
            return require(ReplicatedStorage.Modules.Internals.Skeleton)
        end)
        if ok and skeleton and skeleton.Conduit and type(skeleton.Conduit.Listen) == "function" then
            Conduit = skeleton.Conduit
            print("[UnHub] Conduit found, attaching listeners for _requestEgg and _purchaseEgg")
            -- Listen for events that commonly carry egg info; adjust names if needed
            Conduit.Listen("_requestEgg", function(payload)
                if type(payload) == "table" and payload.data then
                    LastEgg = {
                        id = tostring(payload.data.id or payload.id or payload.dataId or payload.data.id),
                        rarity = tostring(payload.data.rarity or payload.rarity or "Basic"),
                        variety = tostring(payload.data.variety or payload.variety or "Normal")
                    }
                    print(string.format("[UnHub] Conduit _requestEgg -> captured egg: id=%s rarity=%s variety=%s", LastEgg.id, LastEgg.rarity, LastEgg.variety))
                else
                    print("[UnHub] Conduit _requestEgg -> payload missing data or not table")
                end
            end)
            Conduit.Listen("_purchaseEgg", function(payload)
                if type(payload) == "table" and payload.data then
                    LastEgg = {
                        id = tostring(payload.data.id or payload.id or payload.dataId or payload.data.id),
                        rarity = tostring(payload.data.rarity or payload.rarity or "Basic"),
                        variety = tostring(payload.data.variety or payload.variety or "Normal")
                    }
                    print(string.format("[UnHub] Conduit _purchaseEgg -> captured egg: id=%s rarity=%s variety=%s", LastEgg.id, LastEgg.rarity, LastEgg.variety))
                else
                    print("[UnHub] Conduit _purchaseEgg -> payload missing data or not table")
                end
            end)
        else
            warn("[UnHub] Skeleton.Conduit not available or doesn't expose Listen; falling back to conservative scanning")
        end
    end)

    -- Conservative scanner: narrow, low-frequency search for egg info (safe)
    local function conservativeScanForEgg()
        local roots = {}
        if workspace:FindFirstChild("Plots") then table.insert(roots, workspace.Plots) end
        if workspace:FindFirstChild("Shop") then table.insert(roots, workspace.Shop) end
        if workspace:FindFirstChild("EggShop") then table.insert(roots, workspace.EggShop) end
        if #roots == 0 then table.insert(roots, workspace) end

        for _, root in ipairs(roots) do
            for _, obj in ipairs(root:GetChildren()) do
                -- quick checks only on direct children to avoid deep GetDescendants
                local dataChild = obj:FindFirstChild("data")
                if dataChild and dataChild:IsA("StringValue") then
                    local txt = dataChild.Value or ""
                    local jId = txt:match('"id"%s*:%s*"(.-)"') or txt:match("'id'%s*:%s*'(.-)'")
                    local jRarity = txt:match('"rarity"%s*:%s*"(.-)"') or txt:match("'rarity'%s*:%s*'(.-)'")
                    local jVariety = txt:match('"variety"%s*:%s*"(.-)"') or txt:match("'variety'%s*:%s*'(.-)'")
                    if jId and jRarity and jVariety then
                        LastEgg = { id = jId, rarity = jRarity, variety = jVariety }
                        print(string.format("[UnHub] conservativeScanForEgg: found via data child -> id=%s rarity=%s variety=%s", LastEgg.id, LastEgg.rarity, LastEgg.variety))
                        return
                    end
                end
                -- attributes check
                local okId, idAttr = pcall(function() return obj:GetAttribute("id") end)
                if okId and idAttr then
                    local okR, rAttr = pcall(function() return obj:GetAttribute("rarity") end)
                    local okV, vAttr = pcall(function() return obj:GetAttribute("variety") end)
                    if okR and okV and rAttr and vAttr then
                        LastEgg = { id = tostring(idAttr), rarity = tostring(rAttr), variety = tostring(vAttr) }
                        print(string.format("[UnHub] conservativeScanForEgg: found via attributes -> id=%s rarity=%s variety=%s", LastEgg.id, LastEgg.rarity, LastEgg.variety))
                        return
                    end
                end
            end
        end
        print("[UnHub] conservativeScanForEgg: no egg data found in scanned roots")
    end

    -- Build args table in the exact shape required by the remote
    local function buildArgs(egg)
        if not egg then return nil end
        return {
            {
                __raw = true,
                data = {
                    id = egg.id,
                    rarity = egg.rarity,
                    variety = egg.variety
                }
            }
        }
    end

    -- Debounced safe fire
    local lastFire = 0
    local FIRE_COOLDOWN = 1.5
    local function safeFirePurchase()
        if not purchaseRemote then
            purchaseRemote = getPurchaseRemote()
            if not purchaseRemote then
                warn("[UnHub] safeFirePurchase: purchaseRemote not available")
                return false
            end
        end
        if not LastEgg then
            warn("[UnHub] safeFirePurchase: no LastEgg data to purchase")
            return false
        end
        if tick() - lastFire < FIRE_COOLDOWN then
            print("[UnHub] safeFirePurchase: cooldown active, skipping fire")
            return false
        end
        lastFire = tick()
        local args = buildArgs(LastEgg)
        if not args then
            warn("[UnHub] safeFirePurchase: failed to build args")
            return false
        end
        local ok, err = pcall(function() purchaseRemote:FireServer(unpack(args)) end)
        if ok then
            print(string.format("[UnHub] safeFirePurchase: fired purchase for id=%s rarity=%s variety=%s", LastEgg.id, LastEgg.rarity, LastEgg.variety))
        else
            warn("[UnHub] safeFirePurchase: FireServer failed:", err)
        end
        return ok
    end

    -- Single background loop that reacts to toggle state; prevents multiple loops
    local function startAutoBuyLoop()
        if loopRunning then
            print("[UnHub] startAutoBuyLoop: loop already running")
            return
        end
        loopRunning = true
        print("[UnHub] startAutoBuyLoop: starting loop")
        spawn(function()
            while _G.UnHub_AutoBuyEgg do
                -- prefer event-driven data
                if LastEgg then
                    local ok, err = pcall(safeFirePurchase)
                    if not ok then
                        warn("[UnHub] startAutoBuyLoop: safeFirePurchase pcall error:", err)
                    end
                    task.wait(1.5)
                else
                    -- conservative scan at low frequency
                    pcall(conservativeScanForEgg)
                    if LastEgg then
                        pcall(safeFirePurchase)
                    end
                    task.wait(5)
                end
            end
            loopRunning = false
            print("[UnHub] startAutoBuyLoop: loop stopped (toggle disabled)")
        end)
    end

    -- Create the Auto Buy Egg toggle
    local AutoBuyToggle = MainTab:CreateToggle({
        Name = "Auto Buy Egg",
        CurrentValue = false,
        Flag = "AutoBuyEgg",
        Callback = function(Value)
            _G.UnHub_AutoBuyEgg = Value
            if Value then
                print("[UnHub] Toggle enabled: Auto Buy Egg ON")
                -- refresh remote reference
                purchaseRemote = getPurchaseRemote()
                -- start the safe loop
                startAutoBuyLoop()
            else
                print("[UnHub] Toggle disabled: Auto Buy Egg OFF")
                -- disabling will cause the loop to exit naturally
            end
        end,
    })
end
