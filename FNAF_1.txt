-- Minimal UI toggle button (draggable) to show/hide Rayfield
do
    local ST = Instance.new("ScreenGui")
    local T = Instance.new("ImageButton")
    local Corner = Instance.new("UICorner")

    ST.Name = "Toggle UnHub"
    -- Parent to the same UI context Rayfield uses so it appears alongside the Rayfield UI
    -- If Rayfield:GetParent() or its parent is nil, fall back to CoreGui
    local parentGui = nil
    pcall(function()
        parentGui = Rayfield:GetParent() and Rayfield:GetParent().Parent
    end)
    if not parentGui or not parentGui:IsA("Instance") then
        parentGui = game:GetService("CoreGui")
    end
    ST.Parent = parentGui
    ST.ResetOnSpawn = false

    T.Name = "UnHubToggle"
    T.Parent = ST
    T.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    T.Position = UDim2.new(0, 30, 0.08, 0)
    T.Size = UDim2.new(0, 50, 0, 50)
    T.Image = "rbxassetid://112045743772867" -- replace with your decal id if desired
    T.Active = true
    T.Draggable = true
    T.AutoButtonColor = true

    Corner.Name = "Corner"
    Corner.Parent = T
    Corner.CornerRadius = UDim.new(0, 8)

    -- Toggle Rayfield visibility on click
    T.MouseButton1Click:Connect(function()
        local ok, enabled = pcall(function() return Rayfield:GetEnable() end)
        if ok then
            pcall(function() Rayfield:SetEnable(not enabled) end)
        end
    end)
end

pcall(function() Rayfield:LoadConfiguration() end)

-- Egg Status UI (always enabled; toggle removed)
do
    local StatusTab = Window:CreateTab("Egg Status", 4483362458)
    local StatusSection = StatusTab:CreateSection("Spawned Egg Status")
    local StatusParagraph = StatusTab:CreateParagraph({Title = "Egg Status", Content = "Searching for spawned egg..."})

    -- Debug paragraph to show remotes, last args, and last detected egg
    local DebugSection = StatusTab:CreateSection("Auto Debug")
    local DebugParagraph = StatusTab:CreateParagraph({Title = "Auto Debug", Content = "Idle."})

    -- Always enabled; user cannot disable
    local enabled = true

    -- Helper to read a text-like child safely
    local function readTextOrValue(inst)
        if not inst then return nil end
        -- Common GUI text objects
        if inst:IsA("TextLabel") or inst:IsA("TextBox") or inst:IsA("TextButton") then
            return inst.Text
        end
        -- Value objects (StringValue, IntValue, NumberValue, etc.)
        if inst:IsA("ValueBase") and inst.Value ~= nil then
            return tostring(inst.Value)
        end
        -- Try common properties safely
        local ok, txt = pcall(function() return inst.Text end)
        if ok and type(txt) == "string" and txt ~= "" then return txt end
        local ok2, val = pcall(function() return inst.Value end)
        if ok2 and val ~= nil then return tostring(val) end
        return nil
    end

    -- Scans workspace for all children that contain an Eggtag and returns a table of info
    local function getAllSpawnedEggsInfo()
        local results = {}
        for _, child in pairs(workspace:GetChildren()) do
            if child and child:FindFirstChild("Eggtag") then
                local tag = child:FindFirstChild("Eggtag")
                if tag and tag:FindFirstChild("Container") then
                    local container = tag.Container
                    local price = readTextOrValue(container:FindFirstChild("Price")) or "N/A"
                    local rarity = readTextOrValue(container:FindFirstChild("Rarity")) or "N/A"
                    local variety = readTextOrValue(container:FindFirstChild("Variety")) or readTextOrValue(container:FindFirstChild("Mutation")) or "N/A"
                    local eggName = tostring(child.Name or "Unknown")
                    table.insert(results, {Name = eggName, Price = price, Rarity = rarity, Variety = variety, Container = container})
                end
            end
        end
        return results
    end

    -- Formats a list of egg infos into a string for the paragraph
    local function formatEggsForDisplay(eggs)
        if not eggs or #eggs == 0 then
            return "No egg detected."
        end
        local lines = {}
        for i, e in ipairs(eggs) do
            local line = string.format("%d) Name: %s | Price: %s | Rarity: %s | Mutation: %s", i, tostring(e.Name), tostring(e.Price), tostring(e.Rarity), tostring(e.Variety))
            table.insert(lines, line)
        end
        return table.concat(lines, "\n")
    end

    -- Updater loop (always running)
    spawn(function()
        while true do
            if enabled then
                local ok, eggs = pcall(function()
                    return getAllSpawnedEggsInfo()
                end)

                if ok and eggs and #eggs > 0 then
                    local content = formatEggsForDisplay(eggs)
                    pcall(function()
                        StatusParagraph:Set({Title = "Egg Status", Content = content})
                    end)
                else
                    pcall(function()
                        StatusParagraph:Set({Title = "Egg Status", Content = "No egg detected."})
                    end)
                end
            end
            wait(0.5)
        end
    end)

    -- Auto Spawn + Buy Eggs toggle (uses existing getAllSpawnedEggsInfo)
    do
        local AutoSection = StatusTab:CreateSection("Auto Spawn+Buy")
        local autoEnabled = false
        local autoThread = nil

        -- Helper: sanitize name for id (replace spaces and non-alphanum with underscore)
        local function sanitizeNameForId(name)
            if not name then return "Unknown" end
            local s = tostring(name)
            s = s:gsub("%s+", "_")
            s = s:gsub("[^%w_]", "_")
            return s
        end

        -- Remote references (cached) and path strings for debug
        local Rep = game:GetService("ReplicatedStorage")
        local function getRemoteAndPath(name)
            local path = "ReplicatedStorage.Modules.Internals.Skeleton.Conduit.Instances." .. name
            local ok, rem = pcall(function()
                return Rep:WaitForChild("Modules"):WaitForChild("Internals"):WaitForChild("Skeleton"):WaitForChild("Conduit"):WaitForChild("Instances"):WaitForChild(name)
            end)
            if ok and rem then
                return rem, path
            end
            return nil, path
        end
        local requestEggRemote, requestEggPath = getRemoteAndPath("_requestEgg")
        local purchaseEggRemote, purchaseEggPath = getRemoteAndPath("_purchaseEgg")

        -- Helper to update debug UI and print
        local function updateDebug(statusText)
            local content = statusText or "Idle."
            pcall(function()
                DebugParagraph:Set({Title = "Auto Debug", Content = content})
            end)
            pcall(function() print("[Auto Debug] " .. content) end)
        end

        -- Core loop
        local function autoSpawnBuyLoop()
            while autoEnabled do
                -- Show which remotes we have (or not)
                local remInfo = string.format("RequestRemote: %s | PurchaseRemote: %s", requestEggPath or "nil", purchaseEggPath or "nil")
                updateDebug(remInfo)

                -- 1) Request spawn
                if requestEggRemote then
                    local okReq, errReq = pcall(function()
                        local args = {
                            {
                                __raw = true,
                                data = {}
                            }
                        }
                        -- Log the exact args we will send
                        updateDebug(remInfo .. "\nFiring _requestEgg with args: " .. tostring(args))
                        requestEggRemote:FireServer(unpack(args))
                    end)
                    if not okReq then
                        updateDebug(remInfo .. "\n_requestEgg FireServer failed: " .. tostring(errReq))
                    end
                else
                    updateDebug(remInfo .. "\n_requestEgg not found.")
                end

                -- 2) Wait for detection (poll getAllSpawnedEggsInfo)
                local detectedEgg = nil
                local startTime = tick()
                local timeout = 12 -- seconds to wait for an egg to appear
                while autoEnabled and (tick() - startTime) < timeout do
                    local ok, eggs = pcall(function() return getAllSpawnedEggsInfo() end)
                    if ok and eggs and #eggs > 0 then
                        detectedEgg = eggs[1] -- use first detected egg
                        break
                    end
                    wait(0.4)
                end

                if not detectedEgg then
                    updateDebug(remInfo .. "\nNo egg detected within timeout.")
                else
                    -- Show the raw detected fields
                    local detText = string.format("Detected Egg -> Name: %s | Rarity: %s | Variety: %s", tostring(detectedEgg.Name), tostring(detectedEgg.Rarity), tostring(detectedEgg.Variety))
                    updateDebug(remInfo .. "\n" .. detText)

                    -- Extra: attempt to dump container children and their read values for debugging
                    pcall(function()
                        local container = detectedEgg.Container
                        if container and container:GetChildren() then
                            local parts = {}
                            for _, c in ipairs(container:GetChildren()) do
                                local val = readTextOrValue(c) or "<no text/value>"
                                table.insert(parts, tostring(c.Name) .. "=" .. tostring(val))
                            end
                            if #parts > 0 then
                                updateDebug(remInfo .. "\nContainer children: " .. table.concat(parts, "; "))
                            end
                        end
                    end)

                    -- 3) If detected, attempt purchase
                    if purchaseEggRemote then
                        local namePart = sanitizeNameForId(detectedEgg.Name or "Unknown")
                        local rarityPart = tostring(detectedEgg.Rarity or "Basic")
                        local varietyPart = tostring(detectedEgg.Variety or "Glitch")

                        local idString = "entity_" .. namePart .. "_egg"

                        local okBuy, errBuy = pcall(function()
                            local args = {
                                {
                                    __raw = true,
                                    data = {
                                        id = idString,
                                        rarity = rarityPart,
                                        variety = varietyPart
                                    }
                                }
                            }
                            -- Log the exact args we will send to purchase
                            updateDebug(remInfo .. "\nFiring _purchaseEgg with args: id=" .. tostring(idString) .. " | rarity=" .. tostring(rarityPart) .. " | variety=" .. tostring(varietyPart))
                            purchaseEggRemote:FireServer(unpack(args))
                        end)
                        if not okBuy then
                            updateDebug(remInfo .. "\n_purchaseEgg FireServer failed: " .. tostring(errBuy))
                        end
                    else
                        updateDebug(remInfo .. "\n_purchaseEgg not found.")
                    end
                end

                -- 4) Small delay before next cycle to avoid spamming
                local cycleDelay = 1.0
                local waited = 0
                while autoEnabled and waited < cycleDelay do
                    wait(0.2)
                    waited = waited + 0.2
                end
            end
            updateDebug("Auto Spawn+Buy stopped.")
        end

        -- Create the Rayfield toggle
        StatusTab:CreateToggle({
            Name = "Auto Spawn+Buy Eggs",
            CurrentValue = false,
            Flag = "AutoSpawnBuyEggs",
            Callback = function(val)
                autoEnabled = val
                if autoEnabled then
                    -- start loop in a new thread if not already running
                    if not autoThread or coroutine.status(autoThread) ~= "running" then
                        autoThread = spawn(function()
                            autoSpawnBuyLoop()
                        end)
                    end
                else
                    -- disabling will cause loop to exit naturally
                end
            end
        })
    end
end
